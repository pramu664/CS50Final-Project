-- Add gold to the world
Gold = Object:extend()

function Gold:new()
    self.image = love.graphics.newImage("assets/gold1.png")
    self.position_x = love.math.random(100, 600)
    self.position_y = love.math.random(100, 300)
    self.size = love.math.random(3, 4)
end

function Gold:draw()
    love.graphics.draw(self.image, self.position_x, self.position_y, 0, self.size, self.size)
end
